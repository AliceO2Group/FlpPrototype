        Subversion_WC_INFO(${CMAKE_SOURCE_DIR} MY)
        SET(VCS_REVISION "${MY_WC_REVISION}")

        SET(VCS_REVISION "-1")
    endif()

    SET(VCS_REVISION "-1")
ENDIF(Subversion_FOUND)
